 DETAILED PROJECT SCOPE: Hardware-Based Persistent Tracking System with ISP Collaboration
EXECUTIVE SUMMARY
A comprehensive device tracking system that uses permanent hardware identifiers (MAC addresses, motherboard serials) combined with ISP partnership to track and recover stolen laptops and gadgets even after complete OS reinstallation (flushing). The system provides legally admissible evidence for law enforcement and insurance claims.

🎯 PROJECT OBJECTIVES
Objective	Description	Success Metric
Persistent Tracking	Track devices after OS wipe/reinstall	95% identification rate post-flush
IP Independence	Work with dynamic IP addresses	Track devices regardless of IP changes
Legal Evidence	Generate court-admissible proof	Evidence accepted in 90% of cases
Recovery Enablement	Provide actionable data to police	50%+ recovery rate with ISP cooperation
User-Friendly	Simple registration process	< 5 minutes to register a device

class HardwareFingerprinter:
    """
    Installed ONCE on the device before theft
    Captures permanent hardware identifiers
    """
    
    def capture_fingerprint(self):
        return {
            # NETWORK (Most Important - Survives Flush)
            "mac_addresses": {
                "ethernet": self.get_mac("eth0"),      # 00:1A:2B:3C:4D:5E
                "wifi": self.get_mac("wlan0"),         # 00:1A:2B:3C:4D:5F
                "bluetooth": self.get_mac("bt0"),       # 00:1A:2B:3C:4D:60
            },
            
            # MOTHERBOARD (Permanent)
            "motherboard": {
                "serial": self.get_baseboard_serial(),   # CN1296034R00
                "manufacturer": self.get_baseboard_vendor(),  # Dell
                "version": self.get_baseboard_version(),      # 0.1
            },
            
            # BIOS/UEFI (Permanent)
            "bios": {
                "uuid": self.get_bios_uuid(),           # 4C4C4544-0050-4910-8037
                "serial": self.get_bios_serial(),       # ABC123
                "vendor": self.get_bios_vendor(),        # Dell Inc.
                "version": self.get_bios_version(),      # 1.2.5
            },
            
            # STORAGE (Survives Reinstall)
            "drives": [
                {
                    "serial": self.get_drive_serial("C:"),  # WD-WCC123456
                    "model": self.get_drive_model(),        # Samsung EVO 860
                    "size": self.get_drive_size()           # 512GB
                }
            ],
            
            # CPU (Permanent)
            "processor": {
                "id": self.get_cpu_id(),                   # BFEBFBFF000906E9
                "model": self.get_cpu_model(),              # Intel i7-10750H
                "serial": self.get_cpu_serial()             # (if available)
            }
        }

3.2 ISP Integration API

class ISPIntegration:
    """
    Standardized interface for all ISP partners
    """
    
    def __init__(self, isp_name):
        self.isp = isp_name
        self.api_endpoints = {
            "Safaricom": "https://api.safaricom.co.ke/v1/network/mac-lookup",
            "Airtel": "https://api.airtel.africa/v2/device-tracking",
            "Telkom": "https://api.telkom.co.ke/network/active-connections"
        }
        self.api_keys = self.load_api_keys()
    
    def get_current_ip_for_mac(self, mac_address):
        """
        Real-time lookup: What IP is this MAC using NOW?
        """
        response = requests.post(
            self.api_endpoints[self.isp],
            headers={
                "Authorization": f"Bearer {self.api_keys[self.isp]}",
                "Content-Type": "application/json"
            },
            json={
                "mac_address": mac_address,
                "timestamp": datetime.now().isoformat(),
                "request_type": "realtime_lookup"
            }
        )
        
        if response.status_code == 200:
            return {
                "mac": mac_address,
                "current_ip": response.json()["ip_address"],
                "assigned_since": response.json()["lease_start"],
                "connection_type": response.json()["connection_type"],
                "location_area": response.json()["serving_area"]
            }
        return None
    
    def get_ip_history(self, mac_address, days=30):
        """
        Historical lookup: What IPs has this MAC used?
        Critical for building evidence timeline
        """
        response = requests.post(
            f"{self.api_endpoints[self.isp]}/history",
            json={
                "mac_address": mac_address,
                "days": days
            }
        )
        
        return response.json()["ip_history"]  # List of {ip, start_time, end_time}
    
    def get_subscriber_details(self, ip_address, timestamp, warrant_number):
        """
        Legal: Get customer details for specific IP at specific time
        Requires police warrant
        """
        response = requests.post(
            f"{self.api_endpoints[self.isp]}/subscriber-lookup",
            headers={
                "Authorization": f"Bearer {self.api_keys[self.isp]}",
                "Legal-Warrant": warrant_number
            },
            json={
                "ip_address": ip_address,
                "timestamp": timestamp
            }
        )
        
        return {
            "subscriber_name": response.json()["customer_name"],
            "id_number": response.json()["id_number"],
            "physical_address": response.json()["installation_address"],
            "contact_phone": response.json()["phone_number"]
        }

class ISPRealTimeFeed:
    """
    WebSocket connection to ISP for real-time MAC monitoring
    """
    
    def __init__(self):
        self.stolen_macs = self.load_stolen_database()
        self.active_sessions = {}
    
    async def monitor_isp_feed(self, isp_websocket):
        """
        Receive continuous feed of active connections
        """
        async for message in isp_websocket:
            # Message format from ISP:
            # {
            #   "timestamp": "2024-01-15T14:30:45Z",
            #   "connections": [
            #       {
            #           "mac": "00:1A:2B:3C:4D:5E",
            #           "ip": "197.248.65.187",
            #           "device_type": "laptop",
            #           "signal_strength": -45,
            #           "connection_time": "2024-01-15T14:28:00Z"
            #       }
            #   ]
            # }
            
            for connection in message["connections"]:
                if connection["mac"] in self.stolen_macs:
                    # STOLEN DEVICE DETECTED!
                    await self.handle_stolen_sighting(connection)
    
    async def handle_stolen_sighting(self, sighting):
        """
        Silent alert when stolen device connects
        """
        # 1. Log the sighting
        evidence_id = self.log_evidence(sighting)
        
        # 2. Alert owner (silent notification)
        owner = self.get_owner_by_mac(sighting["mac"])
        self.send_owner_alert(owner, sighting)
        
        # 3. If multiple sightings, alert police
        sighting_count = self.get_sighting_count(sighting["mac"], hours=24)
        if sighting_count >= 3:
            self.alert_police(sighting["mac"], evidence_id)

4.1 Continuous Monitoring System

class StolenDeviceMonitor:
    """
    24/7/365 monitoring service
    """
    
    def __init__(self):
        self.isp_feeds = []
        self.stolen_database = StolenDatabase()
        self.evidence_locker = EvidenceLocker()
        
    def run_monitoring_loop(self):
        """
        Main monitoring loop - runs forever
        """
        while True:
            # Check all ISP partners
            for isp in self.isp_feeds:
                # Get all active connections from this ISP
                active_connections = isp.get_active_connections()
                
                for connection in active_connections:
                    if self.is_stolen(connection.mac):
                        self.process_stolen_sighting(connection, isp)
                    
                    # Also check for partial matches (if MAC slightly different)
                    elif self.is_partial_match(connection):
                        self.flag_for_review(connection)
            
            # Sleep 5 minutes between full scans
            # (Real-time alerts come via WebSocket)
            time.sleep(300)
    
    def process_stolen_sighting(self, connection, isp):
        """
        Handle detection of stolen device
        """
        # Create evidence record
        evidence = {
            "device_id": self.get_device_id(connection.mac),
            "mac_address": connection.mac,
            "ip_address": connection.ip,
            "isp": isp.name,
            "timestamp": datetime.now(),
            "location": self.geolocate_ip(connection.ip),
            "connection_duration": connection.duration,
            "signal_strength": connection.signal,
            "confidence": 1.0  # 100% - MAC match
        }
        
        # Store evidence
        evidence_id = self.evidence_locker.store(evidence)
        
        # Update device status
        self.stolen_database.update_last_seen(
            connection.mac, 
            evidence_id
        )
        
        # Notify owner (silent)
        owner = self.get_owner(connection.mac)
        self.notification_service.notify_owner(
            owner_id=owner.id,
            evidence_id=evidence_id,
            silent=True  # Don't alert thief
        )
        
        # Check if we should alert police
        if self.should_alert_police(connection.mac):
            self.police_notification.alert(
                device_mac=connection.mac,
                evidence_ids=self.get_recent_evidence(connection.mac, days=7)
            )

class ConfidenceScorer:
    """
    Calculate confidence that a detected device is the stolen one
    Used when MAC doesn't match exactly (spoofing attempts)
    """
    
    def calculate_confidence(self, detected, stolen_profile):
        """
        Score from 0-1 how likely this is the stolen device
        """
        score = 0.0
        matches = []
        
        # MAC address match (most important)
        if detected.mac == stolen_profile.mac:
            score += 0.5
            matches.append("exact_mac")
        elif self.mac_similarity(detected.mac, stolen_profile.mac) > 0.8:
            score += 0.3  # Possible MAC spoofing attempt
            matches.append("similar_mac")
        
        # ISP range match
        if detected.isp == stolen_profile.home_isp:
            score += 0.15
            matches.append("same_isp")
        
        # Geographic area match
        if self.same_city(detected.location, stolen_profile.home_location):
            score += 0.15
            matches.append("same_area")
        
        # Behavioral patterns
        if self.matches_usage_pattern(detected, stolen_profile):
            score += 0.1
            matches.append("behavior_match")
        
        # Time patterns
        if self.matches_time_pattern(detected.timestamp, stolen_profile):
            score += 0.1
            matches.append("time_pattern")
        
        return {
            "confidence": min(score, 1.0),
            "matches": matches,
            "requires_review": score < 0.7 and score > 0.4
        }

class EvidenceBlockchain:
    """
    Optional: Store evidence hashes on blockchain for tamper-proofing
    """
    
    def record_sighting_on_chain(self, sighting):
        """
        Create immutable record of sighting
        """
        evidence_string = f"{sighting.mac}:{sighting.ip}:{sighting.timestamp}"
        evidence_hash = hashlib.sha256(evidence_string.encode()).hexdigest()
        
        # Store on Ethereum (or other blockchain)
        tx_hash = self.blockchain_contract.addEvidence(
            sighting.device_id,
            evidence_hash,
            sighting.timestamp.isoformat()
        )
        
        return {
            "evidence_hash": evidence_hash,
            "blockchain_tx": tx_hash,
            "block_number": self.get_block_number(tx_hash),
            "timestamp": datetime.now()
        }

class SilentNotifier:
    """
    Ensures thief doesn't know they're being tracked
    """
    
    def notify_owner_silently(self, owner, sighting):
        """
        Send notification without alerting device user
        """
        # Don't send anything to the stolen device
        # Only send to owner's registered contact methods
        
        # Send encrypted email (thief can't access owner's email)
        encrypted = self.encrypt_for_owner(owner.public_key, {
            "type": "sighting",
            "device": sighting.device_id,
            "location": sighting.location,
            "timestamp": sighting.timestamp
        })
        
        self.send_encrypted_email(owner.email, encrypted)
        
        # Send SMS (different device)
        self.send_sms(owner.phone, f"Device {sighting.device_id[:8]}... online. Check app.")
        
        # Log in secure dashboard
        self.add_to_dashboard(owner.id, sighting, silent=True)

class AuditLogger:
    """
    Track all access to sensitive data
    """
    
    def log_access(self, user, resource, action):
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "user_id": user.id,
            "user_role": user.role,
            "user_ip": user.current_ip,
            "resource_type": resource.type,
            "resource_id": resource.id,
            "action": action,
            "authorization_method": user.auth_method,
            "session_id": user.session_id,
            "outcome": "success"
        }
        
        # Store in immutable audit log
        self.audit_database.insert(log_entry)
        
        # Also log to separate secure storage
        self.secure_audit_file.append(log_entry)
        
        # Alert on suspicious access
        if self.is_suspicious(log_entry):
            self.alert_security_team(log_entry)
